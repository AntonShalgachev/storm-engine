
Формат описания итема

[любое имя]

;----------------------------------------------------------------------------------------
;Название итема
;----------------------------------------------------------------------------------------

name=название, являющееся уникальным неизменяемым именем предмета <строка>

;----------------------------------------------------------------------------------------
;Тип итема
;----------------------------------------------------------------------------------------

type=тип <строка>

Типы:
		sword				Основное ручное оружие (сабля)
		pistol				Пистолет
		knife				Нож
		bomb				Бомба
		musket				Мушкет
		healthpotion		Лечилка
		shipshield			Броня для корабля

;----------------------------------------------------------------------------------------
;Описание итема
;----------------------------------------------------------------------------------------

description=описание, отображаемое в интерфейсах <строка>

;----------------------------------------------------------------------------------------
;Картинка итема для интерфейса
;----------------------------------------------------------------------------------------

image=имя картинки, для отображения в интерфейсах <строка>

;----------------------------------------------------------------------------------------
;Модель итема (интерфейс, валяние на земле)
;----------------------------------------------------------------------------------------

model=имя модельки, для отображение в 3D <строка>

;----------------------------------------------------------------------------------------
;Модель итема для использования (на персонаже, корабле)
;----------------------------------------------------------------------------------------

modeluse=имя модельки, для отображение в 3D <строка>

;----------------------------------------------------------------------------------------
;Локатор на котором распологается итем
;----------------------------------------------------------------------------------------

locator=имя локатора на модельке <строка>

;----------------------------------------------------------------------------------------
;Берётся итем автоматически или при нажатии клавиши действия
;----------------------------------------------------------------------------------------

autoget=0 для взятия по кнопке или 1 для автоматического взятия <0 или 1>

;----------------------------------------------------------------------------------------
;Используется итем автоматически при взятии или при опционально
;----------------------------------------------------------------------------------------

autouse=0 для взятия по кнопке или 1 для автоматического взятия <0 или 1>

;----------------------------------------------------------------------------------------
;Уровень использования (крутость предмета)
;----------------------------------------------------------------------------------------

leveluse=1 и более для определения утовня, 0 игнорировать. <целое число>

;----------------------------------------------------------------------------------------
;Коэфициент дистанции использования
;----------------------------------------------------------------------------------------

range=множитель на исходную дистанцию <беззнаковое дробное число>

;----------------------------------------------------------------------------------------
;Коэфициент повреждения от предмета
;----------------------------------------------------------------------------------------

damage=множитель на исходный демедж для врага <беззнаковое дробное число>

;----------------------------------------------------------------------------------------
;Коэфициент защиты
;----------------------------------------------------------------------------------------

shield=множитель на исходный демедж от врага <беззнаковое дробное число>

;----------------------------------------------------------------------------------------
;Количество изменения hp в секунду при использовании
;----------------------------------------------------------------------------------------

dlthp=число в формате <знаковое дробное число>
